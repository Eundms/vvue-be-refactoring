# 커밋 메시지 작성 규칙:
#  - <Tag> : feat, fix, docs, style, refactor, test, chore 등 중 선택
#  - <> : 변경 내용을 간결하게 요약해서 작성 (현재 시제, 50자 이내 권장)
#
# 예)
#   feat: 로그인 기능 추가
#   fix: 회원가입 시 이메일 중복 오류 수정
#
# 본문(선택사항)은 빈 줄로 구분 후 상세 내용 작성 가능

[<Tag>] <변경 요약>

# 본문 (선택사항)
# - 어떤 변경인지 자세히 작성
# - 여러 줄 작성 가능하며 72자 줄바꿈 권장

Resolves: #<이슈번호>
# 'Resolves' 뒤에 관련 이슈 번호를 적으면
# 해당 커밋이 머지될 때 이슈가 자동 닫힙니다.
# 예) Resolves: #123
